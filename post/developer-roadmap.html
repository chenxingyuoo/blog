@charset "UTF-8";
@import "./common/var.css";

@component-namespace el {

  @b alert {
    color: var(--color-text-dark);

    /* 贴边 */
    &.welt {
      border-bottom: 1px solid var(--tab-border-color);
    }

    @e icon {
      color: var(--color-text-dark);
    }

    & .el-alert__description {
      color: var(--color-text-dark);
    }
  }
